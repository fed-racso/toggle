Initialization:

$(document).ready(function() {
    // Nested Tabs 2
    $('#nested-tabs-2').slidetabs();
    
    // Nested Tabs
    $('#nested-tabs').slidetabs();
    
    // Parent Tabs
    $('#parent-tabs').slidetabs({
        contentAnim: ''
    });
});
